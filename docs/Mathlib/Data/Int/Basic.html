    Basic algebraic instances on the integers #

    This file contains instances on ℤ. The stronger one is Int.linearOrderedCommRing.

    source

    CommRing ℤ
    Equations
    • Int.instCommRingInt = CommRing.mk Int.mul_comm

    @[simp]
    theorem Int.cast_id {n : ℤ} :
    ↑n = n
    source
    @[simp]
    theorem Int.cast_mul {α : Type u_1} [NonAssocRing α] (m : ℤ) (n : ℤ) :
    ↑(m * n) = ↑m * ↑n
    source
    theorem Int.cast_Nat_cast {R : Type u_1} {n : ℕ} [AddGroupWithOne R] :
    ↑↑n = ↑n
    source
    @[simp]
    theorem Int.cast_pow {R : Type u_1} [Ring R] (n : ℤ) (m : ℕ) :
    ↑(n ^ m) = ↑n ^ m
